[show source] 
     # File lib/rrd/wrapper.rb, line 113
113:       def info(*args)
114:         argv = to_pointer(["info"] + args)
115:         result = rrd_info(args.size+1, argv)
116:     
117:         info = {}
118:         while result.address != 0
119:           item = RRD::Wrapper::RRDInfo.new result
120:           info[item[:key]] = item[:value][INFO_TYPE[item[:type]].to_sym]
121:           result = item[:next]
122:         end
123:         
124:         return false if info.empty?
125:         info
126:       end
